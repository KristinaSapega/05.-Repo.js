/* Напишите функцию, которая возвращает меньшее из 2 чисел.
 Например, из двух чисел — 8 и 4 — функция должна вернуть 4. 
Если передадим 6 и 6, то функция должна вернуть 6. */

const puzzle = (a, b) => {
  if (a < b) {
    return a;
  } else {
    return b;
  }
};

console.log(puzzle(8, 4));
